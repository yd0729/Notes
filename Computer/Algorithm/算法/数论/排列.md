# 排列

## 排列数 $A_m^n$

```cpp
int numberOfPermutation(int n, int m) {
  int ans = 1;
  while (m--) {
    ans *= n--;
  }
  return ans;
}
```

## 阶乘 $n!$

```cpp
int factorial(int n) {
  int ans = 1;
  while (n) {
    ans *= n--;
  }
  return ans;
}
```

## 求所有排列

```cpp
#include <bits/stdc++.h>

using namespace std;

#define marco_swap(a, b) \
  {                      \
    int temp = a;        \
    a = b;               \
    b = temp;            \
  }  //交换


// 在从 data[l] 开始的 n 个数中选择 m 个数排列
void perm(vector<int> &data, int l, int n, int m, int cnt = 0) {
  if (cnt == m) {
    for (int i = 0; i < m; i++) {
      cout << data[l + i] << (i + 1 == m ? "\n" : " ");
    }
  } else {
    for (int i = l + cnt; i < l + n; i++) {
      marco_swap(data[l + cnt], data[i]);  //交换位置，逐步前提
      perm(data, l, n, m, cnt + 1);
      marco_swap(data[l + cnt], data[i]);  //将位置还回去，对下一次排列负责
    }
  }
}

int main() {
  vector<int> data{0, 1, 2, 3, 4, 5, 6, 8, 9, 10, 12};
  perm(data, 7, 3, 3);
}
```
